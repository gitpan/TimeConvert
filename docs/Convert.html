NAME

        Time::Convert - Interface to converting unix seconds to years, days, hours and minutes.

SYNOPSIS

        use Time::Convert;
        my $convert = new Time::Convert;

EXAMPLE

        use Time::Convert;
        my $convert = new Time::Convert;
           $REPLY   = $convert->ConvertSecs(time);
        print($REPLY);
